Conduit Chat Widget

An embeddable chat widget that connects to your Conduit knowledge graph.

The widget floats in the bottom-right corner. Click the chat bubble to open it. Configure your Conduit API URL and key below, then click "Launch Widget."

Configuration

Script Tag Embed

<script
  src="https://your-conduit/static/widget/conduit-chat.js"
  data-api="https://your-conduit/api/v1"
  data-key="your-api-key"
  data-title="Ask anything"
  data-accent="#8b5cf6"
  data-suggestions="What is Snowflake?|How does Iceberg work?|Compare Spark vs Flink"
  data-persist="true">
</script>

Programmatic API

// Initialize
ConduitChat.init({
  api: 'https://your-conduit/api/v1',
  key: 'your-api-key',
  title: 'Knowledge Assistant',
  accent: '#8b5cf6',
  mode: 'standard',        // or 'swarm'
  position: 'bottom-right', // or 'bottom-left'
  suggestions: ['What is Snowflake?', 'How does Iceberg work?'],
  persist: true,            // save chat history to localStorage
});

// Control
ConduitChat.open();
ConduitChat.close();
ConduitChat.toggle();

// Send a question programmatically
ConduitChat.ask('What is GraphRAG?');

// Clear chat history
ConduitChat.clearHistory();

// Tear down
ConduitChat.destroy();

React / Dojo Integration

import { useEffect } from 'react';

function KnowledgeChat({ apiKey }) {
  useEffect(() => {
    // Load widget script
    const script = document.createElement('script');
    script.src = '/static/widget/conduit-chat.js';
    script.onload = () => {
      window.ConduitChat.init({
        api: process.env.NEXT_PUBLIC_CONDUIT_API,
        key: apiKey,
        title: 'Study Assistant',
        greeting: 'Ask me about any concept!',
      });
    };
    document.body.appendChild(script);
    return () => window.ConduitChat?.destroy();
  }, [apiKey]);

  return null; // Widget manages its own DOM
}
